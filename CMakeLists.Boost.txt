cmake_minimum_required(VERSION 2.8.2)

project(Boost NONE)

include(ExternalProject)
ExternalProject_Add(Boost
  URL               https://github.com/boostorg/boost/archive/boost-1.72.0.tar.gz
  SOURCE_DIR        "${CMAKE_CURRENT_BINARY_DIR}/Boost-src"
  CONFIGURE_COMMAND ${CMAKE_CURRENT_BINARY_DIR}/Boost-src/bootstrap.sh
                    --with-libraries=type_erasure
                    --prefix=${CMAKE_CURRENT_BINARY_DIR}/Boost-build
  BUILD_COMMAND     ${CMAKE_CURRENT_BINARY_DIR}/Boost-src/b2 
                    install --prefix=${CMAKE_CURRENT_BINARY_DIR}/Boost-build
                    variant=release
  BUILD_IN_SOURCE   true
  INSTALL_COMMAND   ""
  TEST_COMMAND      ""
)
